create table public.datalists (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  listitem text null,
  list_type text null,
  user_id uuid null,
  constraint datalists_pkey primary key (id)
) TABLESPACE pg_default;

create unique INDEX IF not exists idx_uniq_datalist on public.datalists using btree (listitem, list_type, user_id) TABLESPACE pg_default;

create index IF not exists datalists_list_type_user_id_idx on public.datalists using btree (list_type, user_id) TABLESPACE pg_default;


create table public.group_members (
  id uuid not null default gen_random_uuid (),
  group_id uuid not null,
  user_id uuid not null,
  role text null default 'member'::text,
  joined_at timestamp without time zone null default now(),
  constraint group_members_pkey primary key (id),
  constraint group_members_group_id_user_id_key unique (group_id, user_id),
  constraint unique_group_members unique (group_id, user_id),
  constraint group_members_group_id_fkey foreign KEY (group_id) references groups (id) on delete CASCADE,
  constraint group_members_user_id_fkey1 foreign KEY (user_id) references profiles (id)
) TABLESPACE pg_default;

create table public.groupjoin_request (
  id uuid not null default gen_random_uuid (),
  group_id uuid null,
  email text not null,
  owner uuid null,
  status text null default 'pending'::text,
  created_at timestamp without time zone null default now(),
  request_by uuid null,
  constraint groupjoin_request_pkey primary key (id),
  constraint groupjoin_request_group_id_fkey foreign KEY (group_id) references groups (id) on delete CASCADE,
  constraint groupjoin_request_owner_fkey foreign KEY (owner) references profiles (id)
) TABLESPACE pg_default;

create table public.groups (
  id uuid not null default gen_random_uuid (),
  name text not null,
  owner uuid null,
  created_at timestamp without time zone null default now(),
  constraint groups_pkey primary key (id),
  constraint groups_owner_fkey foreign KEY (owner) references profiles (id)
) TABLESPACE pg_default;

create table public.invitations (
  id uuid not null default gen_random_uuid (),
  group_id uuid null,
  email text not null,
  invited_by uuid null,
  status text null default 'pending'::text,
  created_at timestamp without time zone null default now(),
  constraint invitations_pkey primary key (id),
  constraint invitations_group_id_fkey foreign KEY (group_id) references groups (id) on delete CASCADE
) TABLESPACE pg_default;

create table public.orders (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  product_id bigint null,
  order_by uuid null default gen_random_uuid (),
  order_to uuid null default gen_random_uuid (),
  status text null default 'pending'::text,
  notes text null default ''::text,
  group_id uuid null default gen_random_uuid (),
  status_date timestamp with time zone null,
  constraint orders_pkey primary key (id),
  constraint orders_group_id_fkey foreign KEY (group_id) references groups (id),
  constraint orders_order_by_fkey foreign KEY (order_by) references profiles (id),
  constraint orders_order_to_fkey foreign KEY (order_to) references profiles (id),
  constraint orders_product_id_fkey foreign KEY (product_id) references products (id)
) TABLESPACE pg_default;

create table public.products (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  name text null,
  image text null,
  price double precision null,
  constraint products_pkey primary key (id)
) TABLESPACE pg_default;

create table public.profiles (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default now(),
  name text null,
  email text null,
  avater text null,
  constraint profiles_pkey primary key (id),
  constraint profiles_id_fkey foreign KEY (id) references auth.users (id) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;

create table public.transactions (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  xndate timestamp with time zone not null default (now() AT TIME ZONE 'utc'::text),
  category text not null,
  item text null,
  amount double precision not null default '0'::double precision,
  notes text null,
  xninout smallint not null default '-1'::smallint,
  xntype text null,
  bankid integer null,
  bank_name text null,
  user_id uuid not null,
  xn_for text null default 'XP'::text,
  xn_group bigint null,
  constraint transactions_pkey primary key (id),
  constraint transactions_user_id_fkey foreign KEY (user_id) references profiles (id)
) TABLESPACE pg_default;

create index IF not exists transactions_xndate_xntype_user_id_xn_for_idx on public.transactions using btree (xndate, xntype, user_id, xn_for) TABLESPACE pg_default;


create table public.transgroup (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  catg_from text null,
  catg_to text null,
  grp_type text null,
  user_id uuid null,
  amount double precision null default '0'::double precision,
  xndate timestamp with time zone null,
  constraint transgroup_pkey primary key (id),
  constraint transgroup_user_id_fkey foreign KEY (user_id) references profiles (id)
) TABLESPACE pg_default;
