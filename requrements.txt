CREATE OR REPLACE FUNCTION get_sumary_by_date(input_date_from DATE,input_date_to DATE,exp_type text)
RETURNS TABLE (expense NUMERIC,income NUMERIC) AS $$
BEGIN
    RETURN QUERY
    SELECT COALESCE(SUM(CASE WHEN xninout = 1 THEN 0 ELSE amount END), 0) AS Expense ,COALESCE(SUM(CASE WHEN xninout = 1 THEN amount ELSE 0 END), 0) AS Income
    FROM transactions
    WHERE xndate >= input_date_from and xndate <= input_date_to
      AND xntype = exp_type;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE OR REPLACE FUNCTION get_XNFOR_balance_prs(userId text,xnfor text)
RETURNS TABLE (item text, balance NUMERIC) AS $$
BEGIN
    RETURN QUERY
    
        select item, sum(amount*xninout) balance from transactions 
        WHERE user_id=userId
        AND xn_for = xnfor
        group by item;
   
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE OR REPLACE FUNCTION get_xnfor_balance(userId text,xnfor text)
RETURNS TABLE (neg NUMERIC,pos NUMERIC) AS $$
BEGIN
    RETURN QUERY
    select COALESCE(sum(case when balance<0 then balance else 0 end),0) as neg,COALESCE(sum(case when balance>0 then balance else 0 end),0) as  pos from
     (
        select item, sum(amount*xninout) balance from transactions 
        WHERE user_id=userId
        AND xn_for = xnfor
        group by item
    ) c;
   
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
